package Entity;

import jakarta.persistence.Entity;
import jakarta.persistence.Table;

@Entity
@Table(name = "Student_Details")

public class Student {
	
	@Column(name = "Student_RollNo")
	@Id
	private int rollNo;
	
	@Column(name = "Student_Name")
	private String name;
	
	@Column(name = "Student_Salary")
	private int salary;
	

}
